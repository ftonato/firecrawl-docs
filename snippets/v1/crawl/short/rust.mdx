let crawlParams = json!({
    "limit": 100,
    "scrapeOptions": {
        "formats": ["markdown", "html"]
    }
});
let crawlResult = app
    .CrawlURL(
        "https://firecrawl.dev",
        Some(crawlParams),
        true,
        2,
        None,
    )
    .await;

match crawlResult {
    Ok(data) => println!("Crawl Result:\n{}", data),
    Err(e) => eprintln!("Crawl failed: {}", e),
}
